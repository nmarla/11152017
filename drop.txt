	   public JsonResult FillModel(string value)
        {
            try
            {
                List<CTSCodeTable> sublstcontent = new List<CTSCodeTable>();
                sublstcontent = _tackingBO.GetItemsfordropdownsfromctscodetable(100);
                ViewData["ComplaintEnquiry"] = sublstcontent;
                List<SelectListItem> subcatlist = new List<SelectListItem>();
                //List<string> items = new List<string>();
                foreach (var item in sublstcontent)
                {
                    subcatlist.Add(new SelectListItem
                    {
                        Text = item.short_description,
                        Value = item.table_code
                    });
                }

                return Json(subcatlist, JsonRequestBehavior.AllowGet);
            }
            catch (Exception ee)
            {
                Console.WriteLine(ee.Message);

                //WriteLog("Exception in Saving Canvas " + txtLocation + " " + ee.Message);
            }

            return null;
        }
